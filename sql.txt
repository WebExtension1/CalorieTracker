    typeID int AUTO_INCREMENT,
    name text,
    PRIMARY KEY (typeID)
);

CREATE TABLE If Not Exists foods (
    foodID int AUTO_INCREMENT,
    name text,
    calories int,
    typeID int,
    PRIMARY KEY (foodID),
    FOREIGN KEY (typeID) REFERENCES types (typeID)
);
